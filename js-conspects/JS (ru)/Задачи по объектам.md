```js
// 1. Создание объекта
const book = {
  title: "1984",
  author: "George Orwell",
  year: 1949,

  // 2. Метод getSummary
  getSummary() {
    return `${this.title} by ${this.author}`;
  }
};

console.log(book.getSummary()); // "1984 by George Orwell"

// 3. Перебор объекта
for (let key in book) {
  if (book.hasOwnProperty(key)) {
    console.log(`${key}: ${book[key]}`);
  }
}

// 4. Object.entries() и forEach()
Object.entries(book).forEach(([key, value]) => {
  console.log(`${key}: ${value}`);
});

// 5. Копия объекта с изменённым year
const updatedBook = { ...book, year: 2025 };
console.log(updatedBook);

// 6. Деструктуризация
const { title, author } = book;
console.log(`Title: ${title}, Author: ${author}`);

// 7. Функция для возврата ключей через запятую
function getKeysAsString(obj) {
  return Object.keys(obj).join(', ');
}

console.log(getKeysAsString(book)); // "title, author, year, getSummary"

// 8. Функция для возврата значений типа string
function getStringValues(obj) {
  return Object.values(obj).filter(value => typeof value === 'string');
}

console.log(getStringValues(book)); // ["1984", "George Orwell"]
```
